#
# Variable
#
dvc_set_env:   
	Usage: $prog <variable> <value>

dvc_set_svn:   
	Usage: $prog <SVN_ROOT> <SVN_URL>

dvc_set_project:   
	Usage: $prog <project>

dvc_set_version:   
	Usage: $prog <version> <stage> <block> <phase>
	Description:
	  Set the default design variable and store the value to .dvc/env/

dvc_get_env:   
	Usage: $prog [--global|--local] [--all] <variable>

dvc_get_svn:   
	Usage: $prog [-v]

dvc_get_project:   
	Usage: $prog [-v]

dvc_get_version:   
	Usage: $prog [-v]

dvc_get_container:   
	Usage: $prog <CONTAINER>

#
# Create Design Project
#
dvc_create_project:   
	Usage: $prog <DESIGN_PROJT>
	Description:
	  create project repository in SVN data

dvc_checkout_project:   
	Usage: $prog <DESIGN_PROJT>
	Description:
	  create a .project folder in current directory
	  checkout project folder to .project/

dvc_set_project:
	Usage: $prog <DESIGN_PROJT>
	Description:
	  set current working project

#
# Create Design Version Folder
#
dvc_create_phase:   
	Usage: $prog <DESIGN_PHASE>
dvc_create_block:   
	Usage: $prog <DESIGN_BLOCK> [<DESIGN_PHASE>]
dvc_create_stage:   
	Usage: $prog <DESIGN_STAGE>
dvc_create_version:   
	Usage: $prog <DESIGN_VERSN> [<DESIGN_STAGE>]

#
# List content under design folder
#
dvc_list_project:   
	Usage: $prog <DESIGN_PROJT>
dvc_list_phase:   
	Usage: $prog <DESIGN_PHASE>
dvc_list_block:   
	Usage: $prog <DESIGN_BLOCK> <DESIGN_PHASE>
dvc_list_stage:   
	Usage: $prog <DESIGN_STAGE>
dvc_list_version:   
	Usage: $prog <DESIGN_VERSN> <DESIGN_STAGE>

#
# Checkout Design Version directory
#
dvc_checkout_phase:   
	Usage: $prog <DESIGN_PHASE>
dvc_checkout_block:   
	Usage: $prog <DESIGN_BLOCK> <DESIGN_PHASE>
dvc_checkout_stage:   
	Usage: $prog <DESIGN_STAGE>
dvc_checkout_version:   
	Usage: $prog <DESIGN_VERSN> <DESIGN_STAGE>
	Description:
	  This command will call dvc_set_version
	checkout version folder to local working directory
	create the following link
		.dvc_block -> .project/
		.dvc_stage -> .project/
		.dvc_version -> .project/
	create ":" link in all four levels

#
# Container
#
dvc_create_container:   
	Usage: $prog <CONTAINER>
	create a container folder in server

dvc_checkout_container:   
	Usage: $prog <CONTAINER>
	checkout container folder to local working directory
	.project/<PROJT>/<PHASE>/<BLOCK>/<STAGE>/<VERSN>
	create a .container link to .project/-/-/-/-/-/<CONTAINER> 

dvc_empty_container:   
	Usage: $prog <CONTAINER>|:
dvc_update_container:   
	Usage: $prog <CONTAINER>|:
dvc_commit_container:   
	Usage: $prog <CONTAINER>|:

dvc_list_container:   
	Usage: $prog <CONTAINER>

dvc_add_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT>
	add object <CONTAINER>/<DESIGN_OBJECT> to SVN repository

dvc_copy_object:   
	Usage: $prog <CONTAINER> <SRC_FILE> [<DEST_NAME>]
dvc_link_object:   
	Usage: $prog <CONTAINER> <SRC_FILE> [<DEST_NAME>]
dvc_rename_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT> <DEST_NAME>
dvc_delete_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT>

#
# Remove
#
dvc_remove_project:   
	Usage: $0:t <DESIGN_PROJT>
dvc_remove_phase:   
	Usage: $0:t <DESIGN_PHASE>
dvc_remove_block:   
	Usage: $0:t <DESIGN_BLOCK> <DESIGN_PHASE>
dvc_remove_stage:   
	Usage: $0:t <DESIGN_STAGE>
dvc_remove_version:   
	Usage: $0:t <DESIGN_VERSN>
dvc_remove_container:   
	Usage: $0:t <DESIGN_VERSN> <DESIGN_STAGE>
