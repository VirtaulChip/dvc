#
# Variable
#
dvc_set_env:   
	Usage: $prog <variable> <value>

dvc_set_svn:   
	Usage: $prog <SVN_ROOT> <SVN_URL>

dvc_set_version:   
	Usage: $prog <DESIGN_VERSN> <DESIGN_STAGE> <DESIGN_BLOCK> <DESIGN_PHASE> <DESIGN_PROJT>


dvc_get_env:   
	Usage: $prog [--global|--local] [--all] <variable>

dvc_get_svn:   
	Usage: $prog [-v]

dvc_get_version:   
	Usage: $prog [-v]

#
# Checkout
#
dvc_checkout_project:   
	Usage: $prog <DESIGN_PROJT>
dvc_checkout_phase:   
	Usage: $prog <DESIGN_PHASE>
dvc_checkout_block:   
	Usage: $prog <DESIGN_BLOCK> <DESIGN_PHASE>
dvc_checkout_stage:   
	Usage: $prog <DESIGN_STAGE>
dvc_checkout_version:   
	Usage: $prog <DESIGN_VERSN> <DESIGN_STAGE>
	checkout version folder to local working directory
	create "-" point to default working path

#
# Container
#
dvc_create_container:   
	Usage: $prog <CONTAINER>
	create a container folder in server

dvc_checkout_container:   
	Usage: $prog <CONTAINER>
	checkout container folder to local working directory
	.project/<PROJT>/<PHASE>/<BLOCK>/<STAGE>/<VERSN>
	create a .container link to .project/-/-/-/-/-/<CONTAINER> 

dvc_empty_container:   
	Usage: $prog <CONTAINER>|:
dvc_update_container:   
	Usage: $prog <CONTAINER>|:
dvc_commit_container:   
	Usage: $prog <CONTAINER>|:

dvc_get_container:   
	Usage: $prog <CONTAINER>
dvc_list_container:   
	Usage: $prog <CONTAINER>
dvc_add_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT>
	add object <CONTAINER>/<DESIGN_OBJECT> to SVN repository

dvc_copy_object:   
	Usage: $prog <CONTAINER> <SRC_FILE> [<DEST_NAME>]
dvc_link_object:   
	Usage: $prog <CONTAINER> <SRC_FILE> [<DEST_NAME>]
dvc_rename_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT> <DEST_NAME>
dvc_delete_object:   
	Usage: $prog <CONTAINER> <DESIGN_OBJECT>
