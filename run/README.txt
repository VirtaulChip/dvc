# ========================================================
# Command Line Execution Example:
# ========================================================
#---------------------------------------------------
# 0. Create Run directory
#---------------------------------------------------

  % mkdir -p RUNDIR/2017_0610
  % cd RUNDIR/2017_0610

  % setenv DOP_HOME $HOME/CAD
  % cp $DOP_HOME/dvc/run/CSHRC  .
  % source CSHRC

  % cp $DOP_HOME/dvc/run/Makefile .
  % make help

#---------------------------------------------------
# 0. Init SVN project repository
#---------------------------------------------------

  % make init

#---------------------------------------------------
# 1. Create design directory in SVN server
#---------------------------------------------------

  % make create_project

#---------------------------------------------------
# 2. Checkout container
#---------------------------------------------------

  % make checkout

#---------------------------------------------------
# 3. Checkin file into container
#---------------------------------------------------

  % make checkin

#---------------------------------------------------
# 4. Update and Commit change into container
#---------------------------------------------------

  % make update_container
  % make commit_container

#---------------------------------------------------
# 5. Clean up project data
#---------------------------------------------------

  % make clean
